// ===== TELEGRAM =====
function sendTele($msg, $token, $chat_id) {
    $url = "https://api.telegram.org/bot$token/sendMessage";
    file_get_contents($url . "?" . http_build_query([
        'chat_id' => $chat_id,
        'text' => $msg
    ]));
}

// ===== FILE DB =====
$db_file = __DIR__ . "/.monitor_db.json";

// ambil data lama
$old = file_exists($db_file) ? json_decode(file_get_contents($db_file), true) : [];

// scan sekarang
$files = scandir($current);
$new = [];

$suspicious = ['shell','backdoor','cmd','upload','eval','php.ini'];

    $path = $current . "/" . $file;

    if (is_file($path)) {

        $new[$file] = filemtime($path);

        // cek file baru
        if (!isset($old[$file])) {

            $report .= "🆕 FILE BARU\n";
            $report .= "📄 $file\n";

            foreach ($suspicious as $key) {
                if (stripos($file, $key) !== false) {
                    $report .= "⚠️ Status: MENCURIGAKAN\n";
                }
            }

            $report .= "\n";
        }

        // cek file diubah
        elseif ($old[$file] != filemtime($path)) {

            $report .= "♻️ FILE DIUBAH\n";
            $report .= "📄 $file\n\n";
        }
    }
}
