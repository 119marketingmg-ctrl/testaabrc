// ===== SCAN RECURSIVE =====
function scanFiles($dir) {
    $result = [];
    $files = scandir($dir);

    foreach ($files as $file) {
        if ($file == "." || $file == "..") continue;

        $path = $dir . "/" . $file;

        if (is_dir($path)) {
            $result += scanFiles($path);
        } else {
            $result[$path] = filemtime($path);
        }
    }

    return $result;
}
